class Person {
    String name;
    boolean isStillAngry;
    boolean isStillNotTalking;
}

public class Apology {

    public static void main(String[] args) {

        Person Arp = new Person();

        Arp.name = "Arp🦋";

        Arp.isStillAngry = true;

        Arp.isStillNotTalking = true;

        if (Arp.isStillAngry && Arp.isStillNotTalking) {

            System.out.println(
                "Dear " + Arp.name + ",\n\n" +

                "I am really sorry. 🥺\n\n" +

                "I tried to stay away because I thought it would be better, " +

                "but sometimes I still worry about you and wonder how you're doing.\n\n" +

                "I don't want to disturb you or make things harder. " +

                "I just miss hearing you, even if it's only for a moment like before.\n\n" +

                "I am truly sorry for that day and for everything that hurt you. " +

                "I still want to send you those snaps before your exams and to wake you up in the morning. " +

                "Please don't be rude to me."
            );

        } else {

            System.out.println(
                "Please unblock me 🥺"
            );

        }

        for (int i = 1; i <= 1; i++) {
            System.out.println("Sorry for that day. I spoke more than I should have.");
        }
    }
}
